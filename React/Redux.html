document.onreadystatechange = function () {
      ReactDOM.render(
        React.createElement('h1', { title: 'Hello React' }, 'React'),
        document.getElementById('app'), () => {
          console.log('ReactDOM render success...')
        }
      )
    }
    console.log(Redux, ReactDOM);
    const { createStore, applyMiddleware, combineReducers } = Redux;
    const ReduxThunk = window.ReduxThunk.default;
  
    function generateID() {
      return Math.random().toString(16).substr(2)
    }
  
    // eg1: 不拆分 reducers
    const initialState = { todos: [] }
    const reducers = (state = initialState, action) => {
      switch (action.type) {
        case 'ADD_TODO':
          return Object.assign({}, state, {
            todos: [...state.todos, action.payload]
          });
        default:
          return state;
      }
    }
    const store = createStore(reducers);
    store.subscribe(() => {
      console.log(store.getState())
    })
    var id = generateID()
    var action = store.dispatch({ type: 'ADD_TODO', payload: { id, name: id.toUpperCase() } })
  
    // eg2: 拆分 reducers, 使用 combineReducers 工具方法组合 reducers
    const todos = (state = [], action) => {
      switch (action.type) {
        case 'ADD_TODO':
          return [...state, action.payload]
        default:
          return state
      }
    }
    const reducers = combineReducers({ todos })
    const store = createStore(reducers)
    store.subscribe(() => {
      console.log(store.getState())
    })
    var id = generateID()
    var action = store.dispatch({ type: 'ADD_TODO', payload: { id, name: id.toUpperCase() } })
  
    // redux-thunk 中间件
    const todos = (state = [], action) => {
      switch (action.type) {
        case 'ADD_TODO':
          return [...state, action.payload]
        default:
          return state
      }
    }
    const reducers = combineReducers({ todos })
    const store = createStore(reducers, applyMiddleware(ReduxThunk));
    store.subscribe(() => {
      console.log(store.getState())
    })
    var id = generateID()
    var action = store.dispatch({ type: 'ADD_TODO', payload: { id, name: id.toUpperCase() } })
    function increment() {
      return {
        type: 'ADD_TODO',
        payload: {
          id: generateID(),
          name: 'Redux-thunk'
        }
      };
    }
  
    function incrementAsync() {
      return (dispatch, getState) => {
        setTimeout(() => {
          dispatch(increment());
        }, 3000);
      };
    }
    var action = store.dispatch(incrementAsync())
